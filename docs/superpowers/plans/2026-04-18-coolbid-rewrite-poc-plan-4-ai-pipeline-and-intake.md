# coolbid-rewrite-poc — Plan 4 of 7: AI Pipeline & Intake

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A contractor can go to `/estimates/new`, describe a job in plain text with optional attachments, hit Submit, wait ~15s, and land on `/estimates/[id]` with a complete priced BOM + labor breakdown + narrative + code callouts — generated by Claude and priced from their catalog. This is the core of the POC. If this doesn't feel right, nothing else matters.

**Architecture:** Client uploads attachments to Supabase Storage, then `POST /api/estimates` with intake payload. Route handler: persists a draft estimate → single Claude Sonnet 4.6 call (BOM + labor + narrative + code callouts in one structured-output response) → deterministic catalog mapping (per-slot lookup, no LLM) → compute totals → persist items + labor lines → return estimate id. Client navigates. Prompt caching keeps repeat calls cheap. No streaming in POC.

**Tech Stack:** Anthropic SDK (Sonnet 4.6 default, env-configurable), Zod for output validation + schema, Vitest for unit tests, Supabase Storage for attachments, Next.js 16 route handlers.

**Spec:** `docs/superpowers/specs/2026-04-18-coolbid-rewrite-poc-design.md` §3 (estimates/bom/labor/share_tokens tables — but share_tokens is deferred to Plan 6), §4 (full AI pipeline), §5.5–5.6 (intake screen + loading state), §7 (schema + mapping tests).

**Plan 3 handoff state:** POC repo `main` at `768f5ba`. Contractors can sign up, onboard, and edit their catalog. Catalog has 35 seeded rows per contractor. Placeholder `/estimates/new` and `/estimates/[id]` pages exist but don't do anything.

**Commit discipline:** feature branch `feature/plan-4-ai-pipeline-and-intake` in POC repo. Plan doc on coolbid `main` via `/commit`. Each task ends in one commit.

**Claude Opus 4.7 note:** The `ANTHROPIC_MODEL` env var defaults to `claude-sonnet-4-6` per spec, but this is env-configurable so you can A/B against Opus 4.7 for hard intakes without code change. The prompt-caching behavior is the same.

```sql
  id uuid primary key default gen_random_uuid(),
```

```ts
const apiKey = process.env.ANTHROPIC_API_KEY?.trim();
if (!apiKey) {
  // Don't throw at import time — some builds (e.g. `next build`) evaluate
  // modules without runtime env. Throw only when the client is actually
  // needed in a request.
  // eslint-disable-next-line no-console
  console.warn("ANTHROPIC_API_KEY is not set; AI pipeline will fail at runtime.");

let client: Anthropic | null = null;

export function getAnthropic(): Anthropic {
  if (client) return client;
  if (!apiKey) throw new Error("ANTHROPIC_API_KEY missing at request time");
  client = new Anthropic({ apiKey });
  return client;
}
```

```ts
// src/lib/ai/schema.ts
import { z } from "zod";
import { SLOT_TAXONOMY, LABOR_CATEGORIES } from "@/lib/catalog/slot-taxonomy";

const SlotEnum = z.enum(SLOT_TAXONOMY);
const UnitEnum = z.enum(["each", "ft", "lb", "job"]);
const TierEnum = z.enum(["good", "better", "best"]);
const LaborCategoryEnum = z.enum(LABOR_CATEGORIES);
const DuctworkStatusEnum = z.enum(["existing_reuse", "new", "modify", "unknown"]);
const SystemTypeEnum = z.enum([
  "ac_only",
  "heat_pump",
  "gas_furnace_ac",
  "dual_fuel",
  "ah_heat_strips",
  "package_unit",
  "other",
]);

export const AiBomItemSchema = z.object({
  slot: SlotEnum,
  name: z.string().trim().min(1).max(200),
  unit: UnitEnum,
  quantity: z.number().nonnegative().max(100000),
  tier_hint: TierEnum.optional(),
  notes: z.string().trim().max(500).optional(),
});

export const AiLaborLineSchema = z.object({
  category: LaborCategoryEnum,
  hours: z.number().nonnegative().max(200),
  notes: z.string().trim().max(500),
});

export const AiParsedJobSpecSchema = z.object({
  system_type: SystemTypeEnum,
  tonnage: z.number().nullable(),
  sqft: z.number().nullable(),
  install_context: z.string().trim().max(1000),
  ductwork_status: DuctworkStatusEnum,
  notes_for_contractor: z.string().trim().max(2000).optional(),
});

export const AiEstimateOutputSchema = z.object({
  parsed_job_spec: AiParsedJobSpecSchema,
  bom_items: z.array(AiBomItemSchema).min(1).max(60),
  labor_lines: z.array(AiLaborLineSchema).min(1).max(20),
  narrative: z.string().trim().min(20).max(4000),
  code_callouts: z.array(z.string().trim().max(500)).max(10),
});

export type AiBomItem = z.infer<typeof AiBomItemSchema>;
export type AiLaborLine = z.infer<typeof AiLaborLineSchema>;
export type AiParsedJobSpec = z.infer<typeof AiParsedJobSpecSchema>;
export type AiEstimateOutput = z.infer<typeof AiEstimateOutputSchema>;
```

```ts
import { AiEstimateOutputSchema } from "../schema";

const valid = {
  parsed_job_spec: {

    tonnage: 3,
    sqft: 1800,
    install_context: "attic air handler swap, tight access",
    ductwork_status: "existing_reuse",

  ],
  narrative: "Replace existing 3-ton split with new condenser and coil. Ductwork stays.",
  code_callouts: [],
};

describe("AiEstimateOutputSchema", () => {
  it("accepts a well-formed minimal output", () => {
    expect(() => AiEstimateOutputSchema.parse(valid)).not.toThrow();
  });

  it("rejects unknown slot enum", () => {
    const bad = { ...valid, bom_items: [{ slot: "not_a_slot", name: "x", unit: "each", quantity: 1 }] };
    expect(() => AiEstimateOutputSchema.parse(bad)).toThrow();
  });

  it("rejects negative quantity", () => {

    expect(() => AiEstimateOutputSchema.parse(bad)).toThrow();
  });

  it("rejects empty narrative", () => {
    const bad = { ...valid, narrative: "" };
    expect(() => AiEstimateOutputSchema.parse(bad)).toThrow();
  });

  it("rejects unknown labor category", () => {
    const bad = { ...valid, labor_lines: [{ category: "Brainstorming", hours: 1, notes: "n" }] };
    expect(() => AiEstimateOutputSchema.parse(bad)).toThrow();
  });

  it("rejects empty bom_items", () => {
    const bad = { ...valid, bom_items: [] };
    expect(() => AiEstimateOutputSchema.parse(bad)).toThrow();
  });
});
```

// src/lib/ai/generate-estimate.ts
import { getAnthropic, DEFAULT_MODEL } from "./anthropic";
import { SYSTEM_PROMPT } from "./prompts/system";
import { AiEstimateOutputSchema, type AiEstimateOutput } from "./schema";
import type { IntakeAttachment } from "@/types/estimate";

type AttachmentPayload = {
  storage_path: string;
  mime: string;
  bytes: Uint8Array; // already downloaded by the caller
};

export type GenerateEstimateInput = {
  intakeText: string;
  attachments: AttachmentPayload[];
};

export type GenerateEstimateResult =
  | { ok: true; output: AiEstimateOutput; usage: { inputTokens: number; outputTokens: number; cacheReadTokens: number } }
  | { ok: false; error: string; rawText?: string };

const MAX_ATTACHMENT_BYTES = 20 * 1024 * 1024; // 20MB per attachment safety cap

function bytesToBase64(bytes: Uint8Array): string {
  // Node runtime (Next server): Buffer is available.
  return Buffer.from(bytes).toString("base64");
}

function attachmentToContentBlock(att: AttachmentPayload):
  | { type: "image"; source: { type: "base64"; media_type: string; data: string } }
  | { type: "document"; source: { type: "base64"; media_type: string; data: string } }
  | null {
  if (att.bytes.byteLength > MAX_ATTACHMENT_BYTES) return null;
  const mt = att.mime.toLowerCase();
  const b64 = bytesToBase64(att.bytes);
  if (mt === "image/jpeg" || mt === "image/png" || mt === "image/webp" || mt === "image/gif") {
    return { type: "image", source: { type: "base64", media_type: mt, data: b64 } };
  }
  if (mt === "application/pdf") {
    return { type: "document", source: { type: "base64", media_type: "application/pdf", data: b64 } };
  }
  return null;
}

function buildUserContent(input: GenerateEstimateInput): Array<Record<string, unknown>> {
  const blocks: Array<Record<string, unknown>> = [];
  for (const att of input.attachments) {
    const block = attachmentToContentBlock(att);
    if (block) blocks.push(block);
  }
  blocks.push({ type: "text", text: input.intakeText.trim() });
  return blocks;
}

async function callClaude(
  input: GenerateEstimateInput,
  retryMessage: string | null,
): Promise<{ text: string; usage: { inputTokens: number; outputTokens: number; cacheReadTokens: number } }> {
  const client = getAnthropic();

  const systemBlocks = [
    {
      type: "text" as const,
      text: SYSTEM_PROMPT,
      cache_control: { type: "ephemeral" as const },
    },
  ];

  const userContent = buildUserContent(input);
  if (retryMessage) {
    userContent.push({
      type: "text",
      text: `Your previous output failed schema validation: ${retryMessage}. Return strictly conformant JSON matching the schema exactly. Output ONLY the JSON object, no prose before or after.`,
    });
  }

  const response = await client.messages.create({
    model: DEFAULT_MODEL,
    max_tokens: 8192,
    system: systemBlocks,
    messages: [
      {
        role: "user",
        // SDK accepts a flexible content array — cast once.
        content: userContent as unknown as Array<{ type: "text"; text: string }>,
      },
    ],
  });

  // Collect text blocks
  const text = response.content
    .filter((b) => b.type === "text")
    .map((b) => (b as { type: "text"; text: string }).text)
    .join("");

  const usage = {
    inputTokens: response.usage.input_tokens ?? 0,
    outputTokens: response.usage.output_tokens ?? 0,
    cacheReadTokens: response.usage.cache_read_input_tokens ?? 0,
  };

  return { text, usage };
}

function stripJsonFences(text: string): string {
  const t = text.trim();
  if (t.startsWith("```")) {
    // ```json ... ```
    const firstNewline = t.indexOf("\n");
    const withoutOpen = firstNewline >= 0 ? t.slice(firstNewline + 1) : t;
    const closed = withoutOpen.replace(/```\s*$/, "");
    return closed.trim();
  }
  return t;
}

type AttemptResult =
  | { kind: "ok"; output: AiEstimateOutput; usage: { inputTokens: number; outputTokens: number; cacheReadTokens: number } }
  | { kind: "parse-error"; rawText: string; usage: { inputTokens: number; outputTokens: number; cacheReadTokens: number } }
  | { kind: "schema-error"; rawText: string; hint: string; usage: { inputTokens: number; outputTokens: number; cacheReadTokens: number } };

async function attemptGenerate(
  input: GenerateEstimateInput,
  retryMessage: string | null,
): Promise<AttemptResult> {
  const { text, usage } = await callClaude(input, retryMessage);
  let candidate: unknown;
  try {
    candidate = JSON.parse(stripJsonFences(text));
  } catch {
    return { kind: "parse-error", rawText: text, usage };
  }
  const parsed = AiEstimateOutputSchema.safeParse(candidate);
  if (parsed.success) {
    return { kind: "ok", output: parsed.data, usage };
  }
  const issue = parsed.error.issues[0];
  const path = issue?.path.join(".") ?? "unknown";
  const msg = issue?.message ?? "invalid shape";
  return { kind: "schema-error", rawText: text, hint: `Zod error at ${path}: ${msg}`, usage };
}

export async function generateEstimate(input: GenerateEstimateInput): Promise<GenerateEstimateResult> {
  // First attempt
  const first = await attemptGenerate(input, null);
  if (first.kind === "ok") return { ok: true, output: first.output, usage: first.usage };

  // Build a hint for the retry based on the failure mode
  const retryHint =
    first.kind === "parse-error"
      ? "output was not valid JSON"
      : first.hint;

  // Single retry
  const second = await attemptGenerate(input, retryHint);
  if (second.kind === "ok") return { ok: true, output: second.output, usage: second.usage };

  const finalError =
    second.kind === "parse-error"
      ? "Claude output was not valid JSON on retry."
      : "Claude output failed schema validation twice.";
  return { ok: false, error: finalError, rawText: second.rawText };
}

```ts
import { SLOT_CUSTOMER_CATEGORY, type Slot } from "./slot-taxonomy";
import type { AiBomItem } from "@/lib/ai/schema";

export type CatalogLookupRow = {
  id: string;
  slot: string;
  customer_category: string;
  tier: "good" | "better" | "best" | null;
  unit: "each" | "ft" | "lb" | "job";
  price_mid: number;
  is_active: boolean;
};

export type MappedBomItem = {
  slot: string;
  customer_category: string;
  catalog_item_id: string | null;
  name: string;
  unit: "each" | "ft" | "lb" | "job";
  quantity: number;
  unit_price: number;
  notes: string | null;
  source: "ai_generated" | "ai_generated_unmapped";
};

// Deterministic picker. Given AI item + active catalog rows for that slot, pick one.
export function chooseCatalogRow(
  aiItem: AiBomItem,
  candidates: CatalogLookupRow[],
): CatalogLookupRow | null {
  const active = candidates.filter((r) => r.is_active);
  if (active.length === 0) return null;

  // If AI gave a tier_hint, prefer an exact-tier match
  if (aiItem.tier_hint) {
    const exact = active.find((r) => r.tier === aiItem.tier_hint);
    if (exact) return exact;
  }

  // Otherwise prefer 'better' tier
  const better = active.find((r) => r.tier === "better");
  if (better) return better;

  // Otherwise: tier-less row if present
  const noTier = active.find((r) => r.tier === null);
  if (noTier) return noTier;

  // Otherwise: lowest price_mid
  return [...active].sort((a, b) => a.price_mid - b.price_mid)[0];
}

export function mapBomItemsToCatalog(
  aiItems: AiBomItem[],
  rowsBySlot: Map<string, CatalogLookupRow[]>,
): MappedBomItem[] {
  return aiItems.map((ai) => {
    const rows = rowsBySlot.get(ai.slot) ?? [];
    const pick = chooseCatalogRow(ai, rows);
    if (pick) {
      return {
        slot: ai.slot,
        customer_category: pick.customer_category,
        catalog_item_id: pick.id,
        name: ai.name,
        unit: ai.unit,
        quantity: ai.quantity,
        unit_price: pick.price_mid,
        notes: ai.notes ?? null,
        source: "ai_generated",
      };
    }
    // Unmapped — fall back to taxonomy's customer_category for the rollup
    const fallbackCategory = SLOT_CUSTOMER_CATEGORY[ai.slot as Slot] ?? "Misc";
    return {
      slot: ai.slot,
      customer_category: fallbackCategory,
      catalog_item_id: null,
      name: ai.name,
      unit: ai.unit,
      quantity: ai.quantity,
      unit_price: 0,
      notes: ai.notes ?? null,
      source: "ai_generated_unmapped",
    };
  });
}
```

```ts
//   total    = subtotal + markup

export type LineForTotals = { quantity: number; unit_price: number };
export type LaborForTotals = { hours: number; rate_per_hour: number };

export type Totals = {
  subtotal_materials: number;
  subtotal_labor: number;
  markup_amount: number;
  total: number;
};

export function computeTotals(
  bomLines: LineForTotals[],
  laborLines: LaborForTotals[],
  marginPct: number,
): Totals {
  const subtotal_materials = round2(
    bomLines.reduce((acc, l) => acc + l.quantity * l.unit_price, 0),
  );
  const subtotal_labor = round2(
    laborLines.reduce((acc, l) => acc + l.hours * l.rate_per_hour, 0),
  );
  const cost = subtotal_materials + subtotal_labor;
  const markup_amount = round2(cost * (marginPct / 100));
  const total = round2(cost + markup_amount);
  return { subtotal_materials, subtotal_labor, markup_amount, total };
}

function round2(n: number): number {
  return Math.round(n * 100) / 100;
}
```

```ts
// src/lib/estimates/persist.ts
"use server";

import type { SupabaseClient } from "@supabase/supabase-js";
import type { AiEstimateOutput } from "@/lib/ai/schema";
import type { MappedBomItem } from "@/lib/catalog/map-to-catalog";
import { computeTotals } from "./totals";
import type { IntakeAttachment } from "@/types/estimate";

export type PersistEstimateInput = {
  contractorId: string;
  customerName: string | null;
  customerAddress: string | null;
  customerEmail: string | null;
  customerPhone: string | null;
  intakeText: string;
  intakeAttachments: IntakeAttachment[];
  aiOutput: AiEstimateOutput;
  mappedBom: MappedBomItem[];
  marginPct: number;
  laborRatePerHour: number;
  customerView: "detailed" | "summary";
};

export async function persistEstimate(
  supabase: SupabaseClient,
  input: PersistEstimateInput,
): Promise<{ estimateId: string }> {
  // Compute totals at creation time
  const totals = computeTotals(
    input.mappedBom.map((b) => ({ quantity: b.quantity, unit_price: b.unit_price })),
    input.aiOutput.labor_lines.map((l) => ({ hours: l.hours, rate_per_hour: input.laborRatePerHour })),
    input.marginPct,
  );

  // Insert the estimate row
  const { data: estimate, error: estimateError } = await supabase
    .from("estimates")
    .insert({
      contractor_id: input.contractorId,
      status: "draft",
      customer_name: input.customerName,
      customer_address: input.customerAddress,
      customer_email: input.customerEmail,
      customer_phone: input.customerPhone,
      intake_text: input.intakeText,
      intake_attachments: input.intakeAttachments,
      parsed_job_spec: input.aiOutput.parsed_job_spec,
      narrative: input.aiOutput.narrative,
      code_callouts: input.aiOutput.code_callouts,
      margin_pct: input.marginPct,
      labor_rate_per_hour: input.laborRatePerHour,
      customer_view: input.customerView,
      subtotal_materials: totals.subtotal_materials,
      subtotal_labor: totals.subtotal_labor,
      markup_amount: totals.markup_amount,
      total: totals.total,
    })
    .select("id")
    .single();
  if (estimateError || !estimate) {
    throw new Error(`persistEstimate: failed to insert estimate: ${estimateError?.message}`);
  }
  const estimateId = estimate.id;

  // Insert bom items
  if (input.mappedBom.length > 0) {
    const bomRows = input.mappedBom.map((b, i) => ({
      estimate_id: estimateId,
      position: i,
      slot: b.slot,
      customer_category: b.customer_category,
      catalog_item_id: b.catalog_item_id,
      name: b.name,
      unit: b.unit,
      quantity: b.quantity,
      unit_price: b.unit_price,
      notes: b.notes,
      source: b.source,
    }));
    const { error: bomError } = await supabase.from("estimate_bom_items").insert(bomRows);
    if (bomError) {
      throw new Error(`persistEstimate: failed to insert bom items: ${bomError.message}`);
    }
  }

  // Insert labor lines
  if (input.aiOutput.labor_lines.length > 0) {
    const laborRows = input.aiOutput.labor_lines.map((l, i) => ({
      estimate_id: estimateId,
      position: i,
      category: l.category,
      hours: l.hours,
      rate_per_hour: input.laborRatePerHour,
      notes: l.notes,
      source: "ai_generated" as const,
    }));
    const { error: laborError } = await supabase.from("estimate_labor_lines").insert(laborRows);
    if (laborError) {
      throw new Error(`persistEstimate: failed to insert labor lines: ${laborError.message}`);
    }
  }

  return { estimateId };
}
```

```ts
// src/app/api/estimates/route.ts
import { NextResponse } from "next/server";
import { z } from "zod";
import { createClient } from "@/lib/supabase/server";
import { generateEstimate } from "@/lib/ai/generate-estimate";
import { mapBomItemsToCatalog, type CatalogLookupRow } from "@/lib/catalog/map-to-catalog";
import { persistEstimate } from "@/lib/estimates/persist";

const IntakeAttachmentSchema = z.object({
  storage_path: z.string().min(1).max(500),
  mime: z.string().min(1).max(80),
  label: z.string().max(200).optional(),
});

const RequestBodySchema = z.object({
  customer_name: z.string().trim().max(200).optional().or(z.literal("")),
  customer_address: z.string().trim().max(500).optional().or(z.literal("")),
  customer_email: z.string().trim().email().optional().or(z.literal("")),
  customer_phone: z.string().trim().max(40).optional().or(z.literal("")),
  intake_text: z.string().trim().min(20, "Please describe the job in at least 20 characters.").max(8000),
  attachments: z.array(IntakeAttachmentSchema).max(10).optional().default([]),
});

async function downloadAttachmentBytes(
  supabase: Awaited<ReturnType<typeof createClient>>,
  storagePath: string,
): Promise<Uint8Array | null> {
  const { data, error } = await supabase.storage.from("attachments").download(storagePath);
  if (error || !data) return null;
  const arrayBuffer = await data.arrayBuffer();
  return new Uint8Array(arrayBuffer);
}

export async function POST(request: Request) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return NextResponse.json({ error: "Not signed in" }, { status: 401 });

  // Parse body
  let body: unknown;
  try {
    body = await request.json();
  } catch {
    return NextResponse.json({ error: "Invalid JSON" }, { status: 400 });
  }
  const parsed = RequestBodySchema.safeParse(body);
  if (!parsed.success) {
    return NextResponse.json(
      { error: "Validation failed", issues: parsed.error.issues },
      { status: 422 },
    );
  }
  const input = parsed.data;

  // Load contractor profile for defaults
  const { data: contractor, error: contractorError } = await supabase
    .from("contractors")
    .select("default_labor_rate, default_margin_pct, default_customer_view")
    .eq("id", user.id)
    .single();
  if (contractorError || !contractor) {
    return NextResponse.json({ error: "Contractor profile missing; complete onboarding first." }, { status: 400 });
  }
  const laborRate = Number(contractor.default_labor_rate ?? 0);
  const marginPct = Number(contractor.default_margin_pct ?? 0);
  const customerView = (contractor.default_customer_view ?? "detailed") as "detailed" | "summary";

  // Download attachments into memory
  const attachmentsPayload = [];
  for (const att of input.attachments ?? []) {
    const bytes = await downloadAttachmentBytes(supabase, att.storage_path);
    if (!bytes) {
      return NextResponse.json({ error: `Attachment not readable: ${att.storage_path}` }, { status: 400 });
    }
    attachmentsPayload.push({ storage_path: att.storage_path, mime: att.mime, bytes });
  }

  // Call Claude
  const ai = await generateEstimate({
    intakeText: input.intake_text,
    attachments: attachmentsPayload,
  });
  if (!ai.ok) {
    return NextResponse.json(
      { error: ai.error, rawText: ai.rawText ?? null },
      { status: 422 },
    );
  }

  // Load contractor's catalog rows grouped by slot (only slots the AI used)
  const emittedSlots = Array.from(new Set(ai.output.bom_items.map((b) => b.slot)));
  const { data: catalogRows, error: catalogError } = await supabase
    .from("contractor_catalog")
    .select("id, slot, customer_category, tier, unit, price_mid, is_active")
    .eq("contractor_id", user.id)
    .in("slot", emittedSlots);
  if (catalogError) {
    return NextResponse.json({ error: `Catalog load failed: ${catalogError.message}` }, { status: 500 });
  }
  const rowsBySlot = new Map<string, CatalogLookupRow[]>();
  for (const r of catalogRows ?? []) {
    const key = r.slot;
    const existing = rowsBySlot.get(key) ?? [];
    existing.push({
      id: r.id,
      slot: r.slot,
      customer_category: r.customer_category,
      tier: (r.tier as CatalogLookupRow["tier"]) ?? null,
      unit: r.unit as CatalogLookupRow["unit"],
      price_mid: Number(r.price_mid),
      is_active: r.is_active,
    });
    rowsBySlot.set(key, existing);
  }

  const mappedBom = mapBomItemsToCatalog(ai.output.bom_items, rowsBySlot);

  // Persist
  let estimateId: string;
  try {
    const result = await persistEstimate(supabase, {
      contractorId: user.id,
      customerName: input.customer_name || null,
      customerAddress: input.customer_address || null,
      customerEmail: input.customer_email || null,
      customerPhone: input.customer_phone || null,
      intakeText: input.intake_text,
      intakeAttachments: input.attachments ?? [],
      aiOutput: ai.output,
      mappedBom,
      marginPct,
      laborRatePerHour: laborRate,
      customerView,
    });
    estimateId = result.estimateId;
  } catch (e) {
    const msg = e instanceof Error ? e.message : "unknown error";
    return NextResponse.json({ error: `Persist failed: ${msg}` }, { status: 500 });
  }

  return NextResponse.json({ id: estimateId, usage: ai.usage }, { status: 200 });
}
```

```tsx
export function IntakeForm({ contractorId }: { contractorId: string }) {
  const router = useRouter();
  const supabase = createClient();
  const fileInputRef = useRef<HTMLInputElement>(null);

  const [customerName, setCustomerName] = useState("");
  const [customerAddress, setCustomerAddress] = useState("");
  const [customerPhone, setCustomerPhone] = useState("");
  const [customerEmail, setCustomerEmail] = useState("");

  const [attachments, setAttachments] = useState<UploadedAttachment[]>([]);

    const files = Array.from(e.target.files ?? []);
    if (files.length === 0) return;
    setUploading(true);
    try {
      for (const file of files) {

          upsert: false,

          continue;
        }
        setAttachments((prev) => [
          ...prev,
          { storage_path: storagePath, mime: file.type, label: file.name },
        ]);

      setUploading(false);
      if (fileInputRef.current) fileInputRef.current.value = "";
```
